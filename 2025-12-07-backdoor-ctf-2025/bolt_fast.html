#!/usr/bin/env python3
import sys
import math
from sympy import primerange
import gmpy2
from gmpy2 import mpz, powmod, gcd

# Load N, e, c from output.txt
with open('output.txt', 'r') as f:
    lines = f.readlines()
    N = int(lines[2].split(' = ')[1].strip())
    e = int(lines[3].split(' = ')[1].strip())
    c = int(lines[4].split(' = ')[1].strip())

print(f'N = {N}')
print(f'e = {e}')
print(f'c = {c}')

# convert to mpz for faster operations
N = mpz(N)
e = mpz(e)
c = mpz(c)

# iterate over 16-bit primes (2^15 to 2^16)
lower = 2**15
upper = 2**16
print(f'Brute forcing dp from {lower} to {upper}')

# Precompute primes using sympy's primerange
primes = list(primerange(lower, upper))
print(f'Number of primes to test: {len(primes)}')

# Use a random base a = 2
a = mpz(2)
for dp in primes:
    dp = mpz(dp)
    # compute a^(dp) mod N
    x = powmod(a, dp, N)
    # compute x^e mod N
    y = powmod(x, e, N)
    # compute gcd(y - a, N)
    g = gcd(y - a, N)
    if g != 1 and g != N:
        print(f'Found factor! dp = {dp}')
        p = g
        q = N // p
        print(f'p = {p}')
        print(f'q = {q}')
        # compute phi = (p-1)*(q-1)
        phi = (p-1)*(q-1)
        d = powmod(e, -1, phi)
        m = powmod(c, d, N)
        flag = int(m)
        # convert to bytes
        flag_bytes = flag.to_bytes((flag.bit_length() + 7) // 8, 'big')
        print(f'Flag: {flag_bytes}')
        with open('.flag', 'wb') as flag_file:
            flag_file.write(flag_bytes)
        with open('.solved', 'w') as solved_file:
            solved_file.write('')
        sys.exit(0)
